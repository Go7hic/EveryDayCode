    <script>

window.onload = function () {
    var aInput = document.getElementsByTagName("input");
    var i = 0;
    for (i = 0; i < aInput.length - 1; i++) {
        aInput[i].onfocus = function () {
            this.className = "text-high"
        };
        aInput[i].onblur = function () {
            this.className = "text"
        }
    }
};
